  --font-mono: ui-monospace, "JetBrains Mono", Consolas, monospace

WCAG contrast notes:
  #e2e8f0 on #0a0a0a = ~13:1 — passes AA and AAA for all text sizes
  #00ff41 on #0a0a0a = ~10.5:1 — passes AA for large text; use for accent/heading only
  #64748b on #0a0a0a = ~3.8:1 — use only for decorative/muted, not readable content

Glitch @keyframes (exact timing — WCAG 2.3.1: ~1.7 state changes/sec at 3s loop, below 3-flash threshold):
  @keyframes glitch — ascending clip-path from bottom to top, offsets on even keyframes
  @keyframes glitch-2 — descending clip-path from top to bottom, 0.15s offset, offsets on odd keyframes

.glitch class pattern:
  Uses data-text attribute for pseudo-element content
  ::before uses cyan (#00ffff) — layer 1
  ::after uses magenta (#ff0055) — layer 2
  Both disabled via @media (prefers-reduced-motion: reduce)

Tailwind v4 globals.css structure (order matters):
  1. @import "tailwindcss"      ← must be first line
  2. @theme { ... }             ← custom properties → Tailwind utilities
  3. @layer base { ... }        ← body, html, ::selection, box-sizing reset
  4. @keyframes ...             ← animation definitions
  5. .glitch class              ← outside any @layer — global CSS class
  6. ::-webkit-scrollbar        ← optional scrollbar styling

    ```css
    @import "tailwindcss";

    /* =========================================================
       DESIGN TOKENS — Tailwind v4 @theme block
       Variables defined here become Tailwind utility suffixes:
       --color-accent → bg-accent, text-accent, border-accent
       --font-mono    → font-mono utility class
    ========================================================= */
    @theme {
      --color-bg: #0a0a0a;
      --color-surface: #111111;
      --color-accent: #00ff41;
      --color-accent-dim: #00cc33;
      --color-text: #e2e8f0;
      --color-text-muted: #64748b;

      --font-mono: ui-monospace, "JetBrains Mono", "Fira Code", Consolas, monospace;
    }

    /* =========================================================
       BASE STYLES
    ========================================================= */
    @layer base {
      *,
      *::before,
      *::after {
        box-sizing: border-box;
        margin: 0;
        padding: 0;
      }

      html {
        overflow-x: hidden;
        /* Smooth scroll fallback for non-Lenis contexts */
        scroll-behavior: auto;
      }

      body {
        background-color: var(--color-bg);
        color: var(--color-text);
        font-family: var(--font-mono);
        /* Prevent layout shift from scrollbar appearing/disappearing */
        scrollbar-gutter: stable;
        /* Subpixel antialiasing — improves text on dark backgrounds */
        -webkit-font-smoothing: antialiased;
        -moz-osx-font-smoothing: grayscale;
      }

      /* Terminal green selection highlight */
      ::selection {
        background-color: var(--color-accent);
        color: var(--color-bg);
      }
    }

    /* =========================================================
       GLITCH KEYFRAMES
       Used by .glitch class below and directly by Phase 2 HeroSection.
       Two layers with offset timing create chromatic aberration effect.

       WCAG 2.3.1 compliance: at a 3s loop with 5 keyframe stops,
       the effective flash rate is ~1.7 state-changes/sec — below
       the 3 flashes/sec threshold (Level A).
    ========================================================= */
    @keyframes glitch {
      0%, 100% { clip-path: inset(0 0 100% 0); transform: translate(0); }
      10%       { clip-path: inset(10% 0 80% 0); transform: translate(-3px, 1px); }
      20%       { clip-path: inset(40% 0 50% 0); transform: translate(3px, -1px); }
      30%       { clip-path: inset(70% 0 20% 0); transform: translate(-2px, 2px); }
      40%       { clip-path: inset(90% 0 5% 0); transform: translate(2px, -2px); }
    }

    @keyframes glitch-2 {
      0%, 100% { clip-path: inset(100% 0 0 0); transform: translate(0); }
      15%       { clip-path: inset(80% 0 15% 0); transform: translate(3px, 1px); }
      25%       { clip-path: inset(50% 0 40% 0); transform: translate(-3px, -1px); }
      35%       { clip-path: inset(20% 0 70% 0); transform: translate(2px, 2px); }
      45%       { clip-path: inset(5% 0 90% 0); transform: translate(-2px, -2px); }
    }

    @keyframes scanline {
      0%   { transform: translateY(-100%); }
      100% { transform: translateY(100vh); }
    }

    @keyframes cursor-blink {
      0%, 100% { opacity: 1; }
      50%       { opacity: 0; }
    }

    /* =========================================================
       GLITCH CSS CLASS
       Usage: <h1 class="glitch" data-text="TITLE">TITLE</h1>

       Two pseudo-element layers use data-text as content and
       animate with offset timing to produce chromatic aberration.

       CRITICAL: Do NOT apply Framer Motion or GSAP transforms to
       a .glitch element — CSS owns the transform on this element.
       If you need Framer Motion entrance, wrap .glitch in a motion
       div and animate the wrapper, not the .glitch element itself.
    ========================================================= */
    .glitch {
      position: relative;
    }

    .glitch::before,
    .glitch::after {
      content: attr(data-text);
      position: absolute;
      inset: 0;
      pointer-events: none;
    }

    /* Layer 1: cyan shift */
    .glitch::before {
      color: #00ffff;
      animation: glitch 3s infinite;
    }

    /* Layer 2: magenta shift, 0.15s offset */
    .glitch::after {
      color: #ff0055;
      animation: glitch-2 3s infinite 0.15s;
    }

    /* Accessibility: disable all motion for users who prefer reduced motion */
    @media (prefers-reduced-motion: reduce) {
      .glitch::before,
      .glitch::after {
        animation: none;
      }
    }

    /* =========================================================
       SCROLLBAR (keeps dark aesthetic consistent across browsers)
    ========================================================= */
    ::-webkit-scrollbar {
      width: 6px;
    }

    ::-webkit-scrollbar-track {
      background: var(--color-bg);
    }

    ::-webkit-scrollbar-thumb {
      background: var(--color-accent-dim);
      border-radius: 3px;
    }

    ::-webkit-scrollbar-thumb:hover {
      background: var(--color-accent);
    }
    ```
